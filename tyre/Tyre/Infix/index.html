val (<|>) : 'a t -> 'b t -> [ `Left of 'a | `Right of 'b ] t

t <|> t' is alt t t'.

val (<&>) : 'a t -> 'b t -> ('a * 'b) t

t <&> t' is seq t t'.

val (*>) : _ t -> 'a t -> 'a t

ti *> t is prefix ti t.

val (<*) : 'a t -> _ t -> 'a t
